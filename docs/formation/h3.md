Cette troisième séance est consacrée à la maîtrise des techniques de prompting avancées pour la génération de contenus textuels structurés et la création d'images professionnelles via l'IA.

## Part 1 : Prompting rédactionnel avancé

### Rappel du framework C-ROFT

Pour obtenir un résultat précis et professionnel d'un LLM (ChatGPT, Gemini, Claude, Mistral), votre prompt doit être structuré :

* **C — Contexte :** Qui êtes-vous ? Dans quelle situation vous trouvez-vous ?
* **R — Rôle :** Quel rôle doit adopter l'IA (ex: *Expert en communication B2B*) ?
* **O — Objectif :** Que doit accomplir le texte (ex: *Rédiger un email de relance prospect*) ?
* **F — Format :** Quelle structure, longueur ou ton adopter (ex: *3 paragraphes, ton professionnel mais chaleureux, objet captivant*) ?
* **T — Target / Cible :** À qui s'adresse ce message ?

<div class="good-reflex-box">
  <h3>Bon reflexe : La méthode itérative</h3>
  <p>Ne vous contentez jamais du premier résultat. Traitez l'IA comme un collaborateur junior :</p>
  <ol>
    <li>Lancez le premier prompt structuré (C-ROFT).</li>
    <li>Analysez le résultat (trop long ? ton trop formel ? manque de clarté ?).</li>
    <li>Affinez par des consignes correctives (<em>« Raccourcis de 30 % », « Utilise des puces », « Adopte un ton plus dynamique »</em>).</li>
  </ol>
</div>

## Part 2 : Prompting visuel (Génération d'images)

### La logique du prompting visuel

Contrairement aux modèles de texte, les générateurs d'images (Midjourney, DALL-E 3, Adobe Firefly) réagissent mieux aux **descriptions factuelles et visuelles** qu'aux consignes abstraites.

<div class="warning-practice-box">
  <h3>Anatomie d'un prompt d'image réussi</h3>
  <ul>
    <li><strong>Sujet principal :</strong> Ce qui doit figurer au centre de l'image (ex: <em>Une poignée de main professionnelle entre deux entrepreneurs</em>).</li>
    <li><strong>Style & Médium :</strong> Photo réaliste, illustration vectorielle, rendu 3D, peinture à l'huile.</li>
    <li><strong>Cadrage & Angle :</strong> Plan rapproché (close-up), vue aérienne, plan large, vue à hauteur d'yeux.</li>
    <li><strong>Éclairage & Ambiance :</strong> Lumière naturelle du matin, éclairage studio doux, tons chauds, ambiance néon.</li>
    <li><strong>Détails techniques :</strong> Profondeur de champ (flou d'arrière-plan / bokeh), 8k, photoréaliste.</li>
  </ul>
</div>

<script>
(function() {
  const questions = [
    {
      task: "Prompt : « Écris-moi un post LinkedIn sur le recrutement. » — Qu'en pensez-vous ?",
      options: [
        "Prompt parfait et direct.",
        "Prompt trop vague : manque de rôle, de cible, de structure et d'angle précis.",
        "Prompt adapté uniquement à Claude."
      ],
      correct: 1,
      explain: "Ce prompt manque de contexte, de contraintes de format, de ton et d'objectif précis. Le résultat sera générique."
    },
    {
      task: "Pour générer une illustration photo-réaliste d'un bureau moderne, quel élément est indispensable dans le prompt visuel ?",
      options: [
        "Demander à l'IA d'être gentille et créative.",
        "Préciser le style (photo), l'éclairage (lumière naturelle) et le cadrage (plan large).",
        "Écrire un paragraphe de 500 mots sur l'histoire de l'entreprise."
      ],
      correct: 1,
      explain: "Les IA de génération d'images ont besoin d'indications descriptives concrètes : style, sujet, éclairage et cadrage."
    },
    {
      task: "Que signifie le 'F' dans le framework C-ROFT ?",
      options: [
        "Fréquence des requêtes",
        "Format (structure, ton, longueur)",
        "Fonctionnalité technique"
      ],
      correct: 1,
      explain: "Le F désigne le Format attendu pour le livrable (ex: liste à puces, tableau, mail de 200 mots, ton chaleureux)."
    }
  ];

  let idx = 0;
  let score = 0;

  const progress = document.getElementById('quiz-progress-h3');
  const taskEl = document.getElementById('quiz-task-h3');
  const choicesEl = document.getElementById('quiz-choices-h3');
  const feedback = document.getElementById('quiz-feedback-h3');
  const nextBtn = document.getElementById('quiz-next-h3');
  const resetBtn = document.getElementById('quiz-reset-h3');
  const scoreEl = document.getElementById('quiz-score-h3');

  function render() {
    if (!taskEl) return;
    feedback.style.display = 'none';
    nextBtn.style.display = 'none';
    resetBtn.style.display = 'none';
    scoreEl.style.display = 'none';

    progress.textContent = "Question " + (idx + 1) + " sur " + questions.length;
    taskEl.textContent = questions[idx].task;
    choicesEl.innerHTML = '';

    questions[idx].options.forEach((opt, oIdx) => {
      const btn = document.createElement('button');
      btn.style.cssText = "text-align:left; padding:10px 14px; border:1px solid var(--md-default-fg-color--lightest, #d5d5d5); border-radius:6px; background:var(--md-default-bg-color, #fff); color:var(--md-typeset-color, #1a1a1a); cursor:pointer; font-size:14px; width:100%; transition: background 0.2s;";
      btn.textContent = opt;
      btn.addEventListener('click', () => answer(oIdx, btn));
      choicesEl.appendChild(btn);
    });
  }

  function answer(oIdx, btn) {
    const q = questions[idx];
    const allBtns = choicesEl.querySelectorAll('button');
    allBtns.forEach(b => b.disabled = true);

    if (oIdx === q.correct) {
      btn.style.background = "rgba(74, 155, 94, 0.2)";
      btn.style.borderColor = "#4a9b5e";
      feedback.style.background = "rgba(74, 155, 94, 0.1)";
      feedback.style.border = "1px solid #4a9b5e";
      feedback.textContent = "✅ Exact ! " + q.explain;
      score++;
    } else {
      btn.style.background = "rgba(201, 86, 74, 0.2)";
      btn.style.borderColor = "#c9564a";
      allBtns[q.correct].style.background = "rgba(74, 155, 94, 0.2)";
      allBtns[q.correct].style.borderColor = "#4a9b5e";
      feedback.style.background = "rgba(201, 86, 74, 0.1)";
      feedback.style.border = "1px solid #c9564a";
      feedback.textContent = "❌ Pas tout à fait. " + q.explain;
    }
    feedback.style.display = 'block';

    if (idx < questions.length - 1) {
      nextBtn.style.display = 'inline-block';
    } else {
      scoreEl.style.display = 'block';
      scoreEl.textContent = "Résultat final : " + score + " / " + questions.length;
      resetBtn.style.display = 'inline-block';
    }
  }

  if (nextBtn) {
    nextBtn.addEventListener('click', () => { idx++; render(); });
  }

  if (resetBtn) {
    resetBtn.addEventListener('click', () => {
      idx = 0;
      score = 0;
      render();
    });
  }

  render();
})();
</script>
