# H5 — Confidentialité, sécurité et sobriété

*Identifier les risques, protéger les données sensibles et adopter une pratique responsable de l'IA.*

---

## Objectifs de cette séance

- **Identifier les données sensibles** (personnelles, financières, confidentielles, RGPD) à bannir des prompts.
- **Distinguer les niveaux de risque** selon les outils et leur hébergement (Cloud vs IA locale).
- **Adopter les réflexes de protection et d'anonymisation** au quotidien.
- **Mesurer l'impact environnemental** de l'usage des grands modèles de langage.

---

## Qu'est-ce qu'une donnée sensible dans un prompt ?

| Type de donnée | Exemples concrets | Risque principal |
| :--- | :--- | :--- |
| **Données personnelles (RGPD)** | Noms, prénoms, emails, téléphones, adresses, NIR | Atteinte à la vie privée, sanctions RGPD |
| **Infos RH & internes** | Salaires, évaluations annuelles, motifs d'absence | Fuite de données confidentielles |
| **Données financières & B2B** | Montants de marchés, marges, prix négociés, SIRET | Perte d'avantage concurrentiel |
| **Secrets d'entreprise** | Clés API, jetons d'accès, code source propriétaire | Risque majeur de cybersécurité |

!!! danger "La règle d'or"
    Si vous ne publieriez pas cette information sur un panneau d'affichage public, **ne la collez pas dans une IA générative grand public**.

    textBox.innerHTML = '';

      const span = document.createElement('span');
      span.textContent = tok.text;
      span.style.cssText = "padding: 3px 6px; border-radius: 4px; cursor: pointer; border: 1px solid transparent; transition: all 0.2s; display: inline-block; margin: 1px 0;";
      
      span.addEventListener('mouseenter', () => {
        if (!verified && !selected.has(idx)) span.style.background = "rgba(26, 95, 180, 0.1)";

      span.addEventListener('mouseleave', () => {
        if (!verified && !selected.has(idx)) span.style.background = "transparent";
      });

      span.addEventListener('click', () => toggle(idx, span));
      textBox.appendChild(span);
    });
  }

  function toggle(idx, span) {
    if (verified) return;
    if (selected.has(idx)) {
      selected.delete(idx);
      span.style.background = "transparent";
      span.style.borderColor = "transparent";
    } else {
      selected.add(idx);
      span.style.background = "rgba(217, 164, 65, 0.3)";
      span.style.borderColor = "#d9a441";
    }
  }

  verifyBtn.addEventListener('click', () => {
    verified = true;
    const tokens = scenarios[currentScenario].tokens;
    const spans = textBox.querySelectorAll('span');
    
    let correctCount = 0;
    let sensitiveTotal = 0;
    let falsePositives = 0;

    tokens.forEach((tok, idx) => {
      if (tok.sensitive) {
        sensitiveTotal++;
        if (selected.has(idx)) {
          spans[idx].style.background = "rgba(74, 155, 94, 0.3)";
          spans[idx].style.borderColor = "#4a9b5e";
          correctCount++;
        } else {
          spans[idx].style.background = "rgba(201, 86, 74, 0.2)";
          spans[idx].style.border = "1px dashed #c9564a";
        }
      } else {
        if (selected.has(idx)) {
          spans[idx].style.background = "rgba(201, 86, 74, 0.3)";
          spans[idx].style.borderColor = "#c9564a";
          falsePositives++;
        }
      }
    });

    results.innerHTML = '';
    tokens.forEach(tok => {
      if (tok.sensitive) {
        const div = document.createElement('div');
        div.style.cssText = "padding: 10px 14px; border-radius: 6px; background: var(--md-default-bg-color, #fff); border: 1px solid var(--md-default-fg-color--lightest, #d5d9de); margin-bottom: 8px; font-size: 13px; color: var(--md-typeset-color, #333);";

        results.appendChild(div);
      }
    });

    results.style.display = 'block';
    summary.style.display = 'block';

    let summaryText = `Résultat : Vous avez identifié ${correctCount} sur ${sensitiveTotal} donnée(s) sensible(s).`;
    if (falsePositives > 0) {
      summaryText += ` (Attention : ${falsePositives} élément(s) non sensible(s) sélectionné(s))`;
    }
    summary.textContent = summaryText;

    verifyBtn.style.display = 'none';
    resetBtn.style.display = 'inline-block';
  });

  resetBtn.addEventListener('click', () => {
    results.style.display = 'none';
    summary.style.display = 'none';
    verifyBtn.style.display = 'inline-block';
    resetBtn.style.display = 'none';
    render();
  });

    currentScenario = parseInt(e.target.value, 10);
    results.style.display = 'none';
    summary.style.display = 'none';
    verifyBtn.style.display = 'inline-block';

  });

## Niveaux de risque et catégories d'outils

| Niveau de risque | Famille d'outils | Utilisation des données |
| :---: | :--- | :--- |
| 🔴 **Élevé** | ChatGPT Gratuit, Gemini Gratuit | Données journalisées et réutilisées pour l'entraînement. |
| 🟠 **Moyen** | ChatGPT Team / Enterprise, Claude Pro | Données chiffrées, non réutilisées pour l'entraînement. |
| 🟢 **Faible** | IA Locales (*Ollama, LM Studio, GPT4All*) | **0% d'envoi réseau**. Traitement 100% local sur la machine. |

<script>
(function() {
  const lengthInput = document.getElementById('eco-length');
  const countInput = document.getElementById('eco-count');
  const modelInput = document.getElementById('eco-model');

  const lengthVal = document.getElementById('eco-length-val');
  const countVal = document.getElementById('eco-count-val');

  const resWater = document.getElementById('res-water');
  const resElec = document.getElementById('res-elec');
  const resCo2 = document.getElementById('res-co2');
  const ecoTip = document.getElementById('eco-tip');

  if (!lengthInput || !countInput || !modelInput) return;

  function calculate() {
    const len = parseInt(lengthInput.value, 10);
    const count = parseInt(countInput.value, 10);
    const factor = parseFloat(modelInput.value);

    lengthVal.textContent = len;
    countVal.textContent = count;

    // Calculs de base pour 1 prompt moyen (50 tokens = ~200 chars)
    // 1 prompt std ≈ 5ml eau, 0.3 Wh, 0.1g CO2
    const tokensRatio = (len / 200);
    const annualPrompts = count * 365;

    const waterLiters = ((annualPrompts * 0.005 * tokensRatio * factor)).toFixed(1);
    const elecKwh = ((annualPrompts * 0.0003 * tokensRatio * factor)).toFixed(2);
    const co2Kg = ((annualPrompts * 0.0001 * tokensRatio * factor)).toFixed(2);

    resWater.textContent = waterLiters + " L";
    resElec.textContent = elecKwh + " kWh";
    resCo2.textContent = co2Kg + " kg";

    if (factor > 2) {
      ecoTip.innerHTML = "💡 <strong>Conseil sobriété :</strong> Pour des tâches simples (correction, résumé), préférez un modèle standard ou local afin d'insérer 3x moins d'impact énergétique.";
    } else if (count > 20) {
      ecoTip.innerHTML = "💡 <strong>Conseil sobriété :</strong> Pensez à regrouper vos demandes (batching) dans un seul prompt structuré plutôt que de faire 20 micro-requêtes.";
    } else {
      ecoTip.innerHTML = "🌱 <strong>Bonne pratique :</strong> Votre usage est modéré. Préférez les réponses courtes et synthétiques pour maintenir cet impact bas.";
    }
  }

  lengthInput.addEventListener('input', calculate);
  countInput.addEventListener('input', calculate);
  modelInput.addEventListener('change', calculate);

  calculate();
})();
</script>
